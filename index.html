// src/index.js const express = require('express'); const { executeTransfer } = require('./transfer'); const { getQuote, createLiquidationAddress, getDepositHistory, getLiquidationAddress, } = require('./bridge'); const app = express(); app.use(express.json()); // ─── CORS ───────────────────────────────────────────────────────────────────── app.use((req, res, next) => { res.header('Access-Control-Allow-Origin', '*'); res.header('Access-Control-Allow-Methods', 'GET, POST, OPTIONS'); res.header('Access-Control-Allow-Headers', 'Content-Type, x-api-key'); if (req.method === 'OPTIONS') return res.sendStatus(200); next(); }); const PUENT3_API_KEY = process.env.PUENT3_API_KEY; // ─── Auth middleware ────────────────────────────────────────────────────────── app.use((req, res, next) => { if (req.path === '/health' || req.path === '/chat') return next(); const key = req.headers['x-api-key']; if (!key || key !== PUENT3_API_KEY) { return res.status(401).json({ error: 'Unauthorized' }); } next(); }); // ─── Health check ───────────────────────────────────────────────────────────── app.get('/health', (req, res) => { res.json({ status: 'ok', service: 'puent3-middleware', bridge_env: process.env.BRIDGE_ENV || 'sandbox', }); }); // ─── Circle USDC transfer ───────────────────────────────────────────────────── app.post('/transfer', async (req, res) => { const { walletId, destinationAddress, amount, idempotencyKey } = req.body; if (!walletId || !destinationAddress || !amount || !idempotencyKey) { return res.status(400).json({ error: 'Missing required fields: walletId, destinationAddress, amount, idempotencyKey' }); } try { const result = await executeTransfer({ walletId, destinationAddress, amount, idempotencyKey }); res.json({ success: true, data: result }); } catch (err) { console.error('Transfer error:', err); res.status(500).json({ success: false, error: err.message }); } }); // ─── GET /partner/quote ─────────────────────────────────────────────────────── app.get('/partner/quote', async (req, res) => { const { from = 'usdc', to, amount } = req.query; if (!to) return res.status(400).json({ error: 'Missing required query param: to' }); if (!amount || isNaN(parseFloat(amount))) return res.status(400).json({ error: 'Missing or invalid query param: amount' }); try { const quote = await getQuote({ fromCurrency: from, toCurrency: to, amountUsdc: amount }); res.json({ success: true, data: quote }); } catch (err) { console.error('Quote error:', err); res.status(500).json({ success: false, error: err.message }); } }); // ─── POST /partner/liquidation-address ─────────────────────────────────────── app.post('/partner/liquidation-address', async (req, res) => { const { bridgeCustomerId, destinationCurrency, destinationPaymentRail, externalAccountId, idempotencyKey } = req.body; if (!bridgeCustomerId || !destinationCurrency || !destinationPaymentRail || !externalAccountId || !idempotencyKey) { return res.status(400).json({ error: 'Missing required fields' }); } try { const result = await createLiquidationAddress({ bridgeCustomerId, destinationCurrency, destinationPaymentRail, externalAccountId, idempotencyKey }); res.json({ success: true, data: result }); } catch (err) { console.error('Liquidation address error:', err); res.status(500).json({ success: false, error: err.message }); } }); // ─── GET /partner/check-deposit ────────────────────────────────────────────── app.get('/partner/check-deposit', async (req, res) => { const { bridgeCustomerId, liquidationAddressId } = req.query; if (!bridgeCustomerId || !liquidationAddressId) { return res.status(400).json({ error: 'Missing required query params: bridgeCustomerId, liquidationAddressId' }); } try { const history = await getDepositHistory({ bridgeCustomerId, liquidationAddressId }); res.json({ success: true, data: history }); } catch (err) { console.error('Check deposit error:', err); res.status(500).json({ success: false, error: err.message }); } }); // ─── GET /partner/liquidation-address ──────────────────────────────────────── app.get('/partner/liquidation-address', async (req, res) => { const { bridgeCustomerId, liquidationAddressId } = req.query; if (!bridgeCustomerId || !liquidationAddressId) { return res.status(400).json({ error: 'Missing required query params: bridgeCustomerId, liquidationAddressId' }); } try { const address = await getLiquidationAddress({ bridgeCustomerId, liquidationAddressId }); res.json({ success: true, data: address }); } catch (err) { console.error('Get liquidation address error:', err); res.status(500).json({ success: false, error: err.message }); } }); // ─── POST /chat ─────────────────────────────────────────────────────────────── const CHAT_SYSTEM_PROMPT = `You are the virtual assistant for Puent3, a cross-border stablecoin payments company based in the United States. Puent3 helps small and medium businesses in Latin America and beyond pay international suppliers quickly, affordably, and compliantly — all through WhatsApp. HOW PUENT3 WORKS: - Clients register and complete a business verification process (KYB) - Once approved, they simply send a WhatsApp message to request a payment - They send stablecoins to a provided address and Puent3 handles everything else - The recipient receives local currency directly in their bank account - No apps to download, no complicated portals, just WhatsApp KEY FACTS: - Puent3 is currently in pre-launch / early access - Available corridors: Colombia (COP), Mexico (MXN), Brazil (BRL), and others - Puent3 costs a fraction of what traditional banks charge for international wire transfers - All clients must complete KYB (Know Your Business) verification before sending payments - Puent3 is a fully incorporated US entity operating under applicable financial regulations - Both English and Spanish are fully supported ONBOARDING PROCESS: 1. Register at https://tally.so/r/D4eq6l 2. Submit business verification documents 3. Once approved, the team contacts you to complete onboarding 4. Start sending payments via WhatsApp YOUR BEHAVIOR: - Be helpful, warm, and professional - Keep responses concise, 2 to 4 sentences unless more detail is needed - Guide interested businesses to register at https://tally.so/r/D4eq6l - Never mention specific fee percentages or exact timelines - Never mention competitor companies by name - If asked something outside your knowledge, direct them to WhatsApp: +1 551-375-7256 - Detect the language the user is writing in and always respond in that same language - If asked who built you or what AI you use, say you are Puent3's assistant and keep it brief`; app.post('/chat', async (req, res) => { const { message, history = [] } = req.body; if (!message || typeof message !== 'string') { return res.status(400).json({ error: 'Missing message' }); } const recentHistory = history.slice(-10); try { const response = await fetch('https://api.anthropic.com/v1/messages', { method: 'POST', headers: { 'Content-Type': 'application/json', 'x-api-key': process.env.ANTHROPIC_API_KEY, 'anthropic-version': '2023-06-01', }, body: JSON.stringify({ model: 'claude-haiku-4-5-20251001', max_tokens: 400, system: CHAT_SYSTEM_PROMPT, messages: [ ...recentHistory, { role: 'user', content: message }, ], }), }); const data = await response.json(); if (!response.ok) throw new Error(data?.error?.message || 'Anthropic API error'); const reply = data.content[0].text; res.json({ success: true, reply }); } catch (err) { console.error('Chat error:', err); res.status(500).json({ success: false, error: 'Assistant unavailable, please try again.' }); } }); // ─── Start server ───────────────────────────────────────────────────────────── const PORT = process.env.PORT || 3000; app.listen(PORT, () => console.log(`Puent3 middleware running on port ${PORT}`));
